//////////////////////////////////////////ACTIVATION WHILE DRAWING/////////////////////////////////////////////
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.EventSystems;

public class DrawableCanvas : MonoBehaviour, IPointerDownHandler, IDragHandler, IPointerUpHandler
{
    public CNNVisualizer cnnVisualizer;
    public int textureSize = 28;
    public int brushSize = 2;
    public RawImage display;
    public Button clearButton;

    [Header("Update Settings")]
    public float updateInterval = 0.15f; // seconds between inference updates while drawing

    private Texture2D drawTexture;
    private Color[] pixels;
    private bool isDrawing = false;
    private float lastUpdateTime = 0f;

    public CNNAnimator cnnAnimator;

    private int lastTexX = -1;
    private int lastTexY = -1;

    void Start()
    {
        drawTexture = new Texture2D(textureSize, textureSize, TextureFormat.RGBA32, false);
        drawTexture.filterMode = FilterMode.Point;
        pixels = new Color[textureSize * textureSize];
        ClearCanvas();
        display.texture = drawTexture;

        if (clearButton != null)
            clearButton.onClick.AddListener(OnClear);
    }

    void OnClear()
    {
        ClearCanvas();
        // Update visualization with blank canvas
        // cnnVisualizer.RunInference(drawTexture);
        cnnAnimator.RunAnimatedInference(drawTexture);
    }

    public void ClearCanvas()
    {
        for (int i = 0; i < pixels.Length; i++)
            pixels[i] = Color.black;
        drawTexture.SetPixels(pixels);
        drawTexture.Apply();
    }

    public void OnPointerDown(PointerEventData eventData)
    {
        isDrawing = true;
        lastTexX = -1;
        lastTexY = -1;
        DrawAtPosition(eventData);
        TryUpdateVisualization();
    }

    public void OnDrag(PointerEventData eventData)
    {
        if (isDrawing)
        {
            DrawAtPosition(eventData);
            TryUpdateVisualization();
        }
    }

    public void OnPointerUp(PointerEventData eventData)
    {
        isDrawing = false;
        lastTexX = -1;
        lastTexY = -1;
        cnnAnimator.RunAnimatedInference(drawTexture);
    }

    void DrawAtPosition(PointerEventData eventData)
    {
        RectTransform rect = display.rectTransform;
        Vector2 localPoint;

        if (!RectTransformUtility.ScreenPointToLocalPointInRectangle(rect, eventData.position, eventData.pressEventCamera, out localPoint))
            return;

        float px = (localPoint.x / rect.rect.width) + 0.5f;
        float py = (localPoint.y / rect.rect.height) + 0.5f;

        int texX = Mathf.Clamp((int)(px * textureSize), 0, textureSize - 1);
        int texY = Mathf.Clamp((int)(py * textureSize), 0, textureSize - 1);

        if (lastTexX >= 0 && lastTexY >= 0)
        {
            // Draw a line from last position to current
            DrawLine(lastTexX, lastTexY, texX, texY);
        }
        else
        {
            // First point, just draw a dot
            PaintBrush(texX, texY);
        }

        lastTexX = texX;
        lastTexY = texY;

        drawTexture.SetPixels(pixels);
        drawTexture.Apply();
    }

    void DrawLine(int x0, int y0, int x1, int y1)
    {
        int dx = Mathf.Abs(x1 - x0);
        int dy = Mathf.Abs(y1 - y0);
        int steps = Mathf.Max(dx, dy);

        if (steps == 0)
        {
            PaintBrush(x0, y0);
            return;
        }

        for (int i = 0; i <= steps; i++)
        {
            float t = (float)i / steps;
            int x = Mathf.RoundToInt(Mathf.Lerp(x0, x1, t));
            int y = Mathf.RoundToInt(Mathf.Lerp(y0, y1, t));
            PaintBrush(x, y);
        }
    }

    void PaintBrush(int centerX, int centerY)
    {
        for (int bx = -brushSize; bx <= brushSize; bx++)
        {
            for (int by = -brushSize; by <= brushSize; by++)
            {
                int finalX = centerX + bx;
                int finalY = centerY + by;

                if (finalX >= 0 && finalX < textureSize && finalY >= 0 && finalY < textureSize)
                {
                    float dist = Mathf.Sqrt(bx * bx + by * by);
                    float strength = Mathf.Clamp01(1f - (dist / (brushSize + 1)));

                    int idx = finalY * textureSize + finalX;
                    float current = pixels[idx].r;
                    float newVal = Mathf.Max(current, strength);
                    pixels[idx] = new Color(newVal, newVal, newVal);
                }
            }
        }
    }

    public Texture2D GetTexture()
    {
        return drawTexture;
    }

    void TryUpdateVisualization()
    {
        if (Time.time - lastUpdateTime >= updateInterval)
        {
            lastUpdateTime = Time.time;
            cnnAnimator.RunAnimatedInference(drawTexture);
        }
    }
}
